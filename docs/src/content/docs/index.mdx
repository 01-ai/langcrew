### ðŸ’¡ Best Developer Experience

```python
from langcrew import CrewBase, agent, task
from langcrew_tools import WebSearchTool

@CrewBase
class MarketingCrew:
    agents_config = "config/agents.yaml"
    tasks_config = "config/tasks.yaml"
    
    @agent
    def researcher(self):
        return Agent(
            config=self.agents_config["researcher"],
            tools=[WebSearchTool()]
        )
    
    @task  
    def market_analysis(self):
        return Task(
            config=self.tasks_config["market_analysis"],
            agent=self.researcher()
        )

# Launch web service in 3 lines
from langcrew.web import create_server
server = create_server(MarketingCrew().crew())
server.run()  # Visit http://localhost:8000/docs
```
